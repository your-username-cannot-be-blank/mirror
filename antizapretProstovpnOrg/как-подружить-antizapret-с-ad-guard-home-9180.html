Сделал поддержку внешних DNS через ENV переменную + сделал пример docker-compose с общей сетью с adguard контейнером:

github.com

xtrime-ru/antizapret-vpn-docker/blob/master/docker-compose.adguard.yml

services:
  antizapret-vpn:
    extends:
      file: docker-compose.base.yml
      service: antizapret-vpn
    networks:
      static-network:
        ipv4_address: 10.5.0.2
    environment:
      - DNS=10.5.0.3
  adguardhome:
    image: adguard/adguardhome
    container_name: adguardhome
    ports:
      - 53:53/tcp
      - 53:53/udp
      - 784:784/udp
      - 853:853/tcp
      - 3000:3000/tcp
      - 80:80/tcp
